import os

def search(path,file_name):
    try:
      files = os.listdir(path)
      trunk = 0
    except OSError:
      print "Access Denied: " + path
      trunk = 1

    if trunk == 0:
      for f in files:
        bool = os.path.isdir(path + f)
        if bool == True:
          search(path + f + '/',file_name)
        else:
          if file_name == f:
            print 'Found : ' + path + f
